38 //----------------------------------------------------------------------------
39 //
40 // ImathMath.h
41 //
42 // This file contains template functions which call the double-
43 // precision math functions defined in math.h (sin(), sqrt(),
44 // exp() etc.), with specializations that call the faster
45 // single-precision versions (sinf(), sqrtf(), expf() etc.)
46 // when appropriate.
47 //
48 // Example:
49 //
50 // double x = Math<double>::sqrt (3); // calls ::sqrt(double);
51 // float y = Math<float>::sqrt (3); // calls ::sqrtf(float);
52 //
53 // When would I want to use this?
54 //
55 // You may be writing a template which needs to call some function
56 // defined in math.h, for example to extract a square root, but you
57 // don't know whether to call the single- or the double-precision
58 // version of this function (sqrt() or sqrtf()):
59 //
60 // template <class T>
61 // T
62 // glorp (T x)
63 // {
64 // return sqrt (x + 1); // should call ::sqrtf(float)
65 // } // if x is a float, but we
66 // // don't know if it is
67 //
68 // Using the templates in this file, you can make sure that
69 // the appropriate version of the math function is called:
70 //
71 // template <class T>
72 // T
73 // glorp (T x, T y)
74 // {
75 // return Math<T>::sqrt (x + 1); // calls ::sqrtf(float) if x
76 // } // is a float, ::sqrt(double)
77 // // otherwise
78 //
79 //----------------------------------------------------------------------------

153 template <class T>
154 inline T
155 sinx_over_x (T x)
156 {
157  if (x * x < limits<T>::epsilon())
158  return T (1);
159  else
160  return Math<T>::sin (x) / x;
161 }

163 //--------------------------------------------------------------------------
164 // Compare two numbers and test if they are "approximately equal":
165 //
166 // equalWithAbsError (x1, x2, e)
167 //
168 // Returns true if x1 is the same as x2 with an absolute error of
169 // no more than e,
170 //
171 // abs (x1 - x2) <= e
172 //
173 // equalWithRelError (x1, x2, e)
174 //
175 // Returns true if x1 is the same as x2 with an relative error of
176 // no more than e,
177 //
178 // abs (x1 - x2) <= e * x1
179 //
180 //--------------------------------------------------------------------------
181 
182 template <class T>
183 inline bool
184 equalWithAbsError (T x1, T x2, T e)
185 {
186  return ((x1 > x2) ? x1 - x2 : x2 - x1) <= e;
187 }
188 
189 template <class T>
190 inline bool
191 equalWithRelError (T x1, T x2, T e)
192 {
193  return ((x1 > x2) ? x1 - x2 : x2 - x1) <= e * ((x1 > 0) ? x1 : -x1);
194 }
